<script>
    const quizData = [
        {
            q: "在派對中，你通常會？",
            options: [
                { text: "主動與陌生人聊天", score: 1 },
                { text: "待在熟人身邊或角落", score: 0 }
            ]
        },
        {
            q: "放假時，你更傾向？",
            options: [
                { text: "出門找朋友聚會", score: 1 },
                { text: "獨自留在家看書或追劇", score: 0 }
            ]
        },
        {
            q: "遇到問題時，你習慣？",
            options: [
                { text: "找人討論，抒發想法", score: 1 },
                { text: "先自己安靜思考解決方案", score: 0 }
            ]
        }
    ];

    let currentIdx = 0;
    let totalScore = 0;

    function loadQuestion() {
        const currentQuiz = quizData[currentIdx];
        document.getElementById('progress').innerText = `問題 ${currentIdx + 1} / ${quizData.length}`;
        document.getElementById('question-text').innerText = currentQuiz.q;
        const optionsDiv = document.getElementById('options-container');
        optionsDiv.innerHTML = '';

        currentQuiz.options.forEach(opt => {
            const btn = document.createElement('button');
            btn.innerText = opt.text;
            btn.onclick = () => selectOption(opt.score);
            optionsDiv.appendChild(btn);
        });
    }

    function selectOption(score) {
        totalScore += score;
        currentIdx++;
        if (currentIdx < quizData.length) {
            loadQuestion();
        } else {
            showResult();
        }
    }

    function showResult() {
        document.getElementById('quiz-content').style.display = 'none';
        const resultDiv = document.getElementById('result');
        const resultText = document.getElementById('result-text');
        resultDiv.style.display = 'block';

        if (totalScore >= 2) {
            resultText.innerText = "你是一個活潑的外向者！社交對你來說是充電。";
        } else {
            resultText.innerText = "你是一個深沉的內向者！你更享受安靜的私人時光。";
        }
    }

    loadQuestion();
</script>
